# Excluding trials with no blink:
if(cond %in% c('cond-v', 'cond-m')){
  excl_no_blink_trials = !trials$trial %in% blanks$trial
  if(sum(excl_no_blink_trials)){  # if there is one or more such trials
    excl_no_blink = data.frame(trial = trials$trial[!trials$trial %in% blanks$trial],
                               excl_reason = 'no blink')
  } else {
    excl_no_blink = NULL
  }
} else {
  excl_no_blink = NULL
}

# Excluding trials with target blinks (blinks during target presentation):
if(cond!='cond-m'){
  merged = merge(trials[,c('trial', 'cue_time', 'resp_time', 'trial_time_beg')],
                 blanks[,c('trial', 'blank_time_beg', 'blank_time_end')], by='trial')
  # Reaction time is the response time stamp minus trial start time stamp:
  merged$rt = with(merged, resp_time - trial_time_beg)
  # Finding blanks that occur more than 'eyes open' tolerance post-target presentation.
  excl_targ_blink = data.frame(trial = unique(merged$trial[merged$blank_time_end>
                                                             (merged$rt+eo_tolerance)]),
                               excl_reason = 'target blink')
} else {
  excl_targ_blink = NULL
}
